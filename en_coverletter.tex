% ============================================================================
% Class import
% "Conventionnal" option is for conventionnal letters.
% "CoverLetter" option is for cover letters.
% Two languages are available : "fr" and "en"

\documentclass[CoverLetter, en]{lettertemplate}

% ============================================================================
% Constant definition
% *1: Those constants must be either both empty or both non-empty.
% *2: At least Description or FirstName/LastName must be given.

\def\SenderFirstName{FirstName}         % *1
\def\SenderLastName{LastName}           % *1
\def\SenderAddress{Address}
\def\SenderCity{City}
\def\SenderCityCode{XXXXX}
\def\SenderPhone{Phone number}          % Optionnal
\def\SenderEmail{Email}                 % Optionnal

\def\ReceiverFirstName{FirstName}       % *1 *2
\def\ReceiverLastName{LastName}         % *1 *2
\def\ReceiverDescription{Position}      % *2
\def\ReceiverCompany{Company}
\def\ReceiverAddress{Address}
\def\ReceiverCityCode{XXXXX}
\def\ReceiverCity{City}

\def\Object{}
\def\Attachment{}                       % Optionnal

% ============================================================================
% Letter content

\begin{document}

    \makeheader

    \noindent
    Dear ...,

    \paragraph{}
        % Introduction
        ...

    \paragraph{}
        % Content
        ...

    \vskip 1cm

    \noindent
    Yours sincerly,

    \sign

\end{document}
